function displayAdjacencyMatrix(matrix, vertices)
{
	foreach(vertex; vertices.iterator())
		println("" + vertex.__id + " ");


	it = vertices.iterator();

	foreach(line; matrix.iterator())
	{
		if(it.hasNext())
			vertex = it.next();

		println("" + vertex.__id + " |");
		foreach(item; line.iterator())
			println("" + item + " ");

		println("\n");
	}
}
